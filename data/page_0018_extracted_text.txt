❖ Objekat (object) je primerak (instanca) klase
Clock* pClk1 = new Clock;
pClk1->setTime(10,12,0);
Clock* pClk2 = new Clock;
pClk2->setTime(23,25,36);
pClk1->tick();
pClk2->tick();
cout << pClk1->getTime() << endl;
delete pClk1;
cout << pClk2->getTime() << endl;
delete pClk2;
