template <class T>
class RecycleBin {
public:
  void* alloc () {
    Slot* p=head;
    if (p) head=p->next; else p = new Slot;
    return p;
  }
  void  free (void* addr) { head = new (addr) Slot(head); }
private:
  struct Slot {
    Slot (Slot* nxt=nullptr) : next(nxt) {}
    union {
      Slot* next;
      char slot[sizeof(T)];
    };
  };
  Slot* head = nullptr;
};
